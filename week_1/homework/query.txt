4) Which was the day with the largest trip distance Use the pick up time for your calculations.

WITH date_table AS (
	SELECT lpep_pickup_datetime::date as date,
			trip_distance as distance
	FROM test_table5)
	
	Select date, Max(distance) as DIST
	FROM date_table
	GROUP BY date
	ORDER by DIST desc
Ans= 2019-01-15
